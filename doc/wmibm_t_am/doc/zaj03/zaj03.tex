\textbf{Zadanie 2} Oblicz granice jednostronne:\\

\textbf{(b)} $\lim\limits_{x\rightarrow 1^{-}} \frac{x^2+1}{x^2-3 x+2}$, $\lim\limits_{x\rightarrow 1^{+}} \frac{x^2+1}{x^2-3 x+2}$, $\lim\limits_{x\rightarrow 2^{-}} \frac{x^2+1}{x^2-3 x+2}$, $\lim\limits_{x\rightarrow 2^{+}} \frac{x^2+1}{x^2-3 x+2}$,\\

$\lim\limits_{x\rightarrow 1^{+}} e^{\frac{x}{x^2-1}}$, $\lim\limits_{x\rightarrow -1^{-}} e^{\frac{x}{x^2-1}}$, $\lim\limits_{x\rightarrow -1^{+}} e^{\frac{x}{x^2-1}}$, 
\textbf{(d)} $\lim\limits_{x\rightarrow -1^{-}} \ln\frac{x}{1-x^2}$, 
$\lim\limits_{x\rightarrow 0^{+}} \ln\frac{x}{1-x^2}$, $\lim\limits_{x\rightarrow 1^{-}} \ln\frac{x}{1-x^2}$.\\
